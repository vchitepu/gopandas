> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a CLI binary `gopandas` with `read` and `convert` subcommands for processing data files from the terminal.

**Architecture:** Binary at `cmd/gopandas/main.go` using cobra. Each subcommand in its own file. File format inferred from extension. Operations compose in a pipeline: load -> select -> filter -> groupby/agg -> sort -> output.

**Tech Stack:** Go 1.21+, github.com/spf13/cobra

**Depends on:** All gopandas packages (dtype, index, arrowutil, series, dataframe, groupby, ops, dataio)

### Step 1.3: Create format helper — infer format from file extension

- [ ] Create `cmd/gopandas/format.go`

**File:** `cmd/gopandas/format.go`

```go
package main

import (
	"fmt"
	"os"
	"path/filepath"
	"strings"

	"github.com/vchitepu/gopandas/dataframe"
	csvio "github.com/vchitepu/gopandas/dataio/csv"
	jsonio "github.com/vchitepu/gopandas/dataio/json"
	parquetio "github.com/vchitepu/gopandas/dataio/parquet"
)

// inferFormat returns "csv", "json", or "parquet" based on extension.
func inferFormat(path string) (string, error) {
	ext := strings.ToLower(filepath.Ext(path))
	switch ext {
	case ".csv":
		return "csv", nil
	case ".json":
		return "json", nil
	case ".parquet", ".pq":
		return "parquet", nil
	default:
		return "", fmt.Errorf("unsupported file extension: %s", ext)
	}
}

// loadFile reads a file into a DataFrame. format is "csv", "json", or "parquet".
func loadFile(path, format string) (dataframe.DataFrame, error) {
	switch format {
	case "csv":
		f, err := os.Open(path)
		if err != nil {
			return dataframe.DataFrame{}, err
		}
		defer f.Close()
		return csvio.FromCSV(f)
	case "json":
		f, err := os.Open(path)
		if err != nil {
			return dataframe.DataFrame{}, err
		}
		defer f.Close()
		return jsonio.FromJSON(f)
	case "parquet":
		return parquetio.FromParquet(path)
	default:
		return dataframe.DataFrame{}, fmt.Errorf("unsupported format: %s", format)
	}
}

// writeFile writes a DataFrame to a file. format is "csv", "json", or "parquet".
func writeFile(df dataframe.DataFrame, path, format string) error {
	f, err := os.Create(path)
	if err != nil {
		return err
	}
	defer f.Close()

	switch format {
	case "csv":
		return csvio.ToCSV(df, f)
	case "json":
		return jsonio.ToJSON(df, f)
	case "parquet":
		return parquetio.ToParquet(df, f)
	default:
		return fmt.Errorf("unsupported format: %s", format)
	}
}
```

### Step 1.5: Create read subcommand — basic file loading + head display

- [ ] Create `cmd/gopandas/read.go` with initial read command (loads file, prints head)

**File:** `cmd/gopandas/read.go`

```go
package main

import (
	"fmt"
	"os"
	"strings"

	"github.com/spf13/cobra"
	"github.com/vchitepu/gopandas/dataframe"
)

var (
	readHead     int
	readTail     int
	readDescribe bool
	readShape    bool
	readDTypes   bool
	readSelect   string
	readFilter   string
	readGroupBy  string
	readAgg      string
	readSort     string
	readSortDesc bool
	readOutput   string
	readFormat   string
)

var readCmd = &cobra.Command{
	Use:   "read <file>",
	Short: "Read and display a data file",
	Long:  "Read a CSV, JSON, or Parquet file and apply operations (filter, select, groupby, sort, etc.)",
	Args:  cobra.ExactArgs(1),
	RunE:  runRead,
}

func init() {
	readCmd.Flags().IntVar(&readHead, "head", 0, "Print first N rows (default: 5 if no other display flag set)")
	readCmd.Flags().IntVar(&readTail, "tail", 0, "Print last N rows")
	readCmd.Flags().BoolVar(&readDescribe, "describe", false, "Print descriptive statistics")
	readCmd.Flags().BoolVar(&readShape, "shape", false, "Print (rows, cols)")
	readCmd.Flags().BoolVar(&readDTypes, "dtypes", false, "Print column names and dtypes")
	readCmd.Flags().StringVar(&readSelect, "select", "", "Select columns (comma-separated)")
	readCmd.Flags().StringVar(&readFilter, "filter", "", "Filter rows by expression (e.g. \"price > 100\")")
	readCmd.Flags().StringVar(&readGroupBy, "groupby", "", "Group by column")
	readCmd.Flags().StringVar(&readAgg, "agg", "", "Aggregation function: sum, mean, count, min, max")
	readCmd.Flags().StringVar(&readSort, "sort", "", "Sort by column")
	readCmd.Flags().BoolVar(&readSortDesc, "sort-desc", false, "Sort descending (used with --sort)")
	readCmd.Flags().StringVar(&readOutput, "output", "", "Write result to file")
	readCmd.Flags().StringVar(&readFormat, "format", "", "Output format: csv, json, parquet (overrides extension)")

	rootCmd.AddCommand(readCmd)
}

func runRead(cmd *cobra.Command, args []string) error {
	path := args[0]

	// Infer input format from extension.
	format, err := inferFormat(path)
	if err != nil {
		return err
	}

	// Load the file.
	df, err := loadFile(path, format)
	if err != nil {
		return fmt.Errorf("failed to read %s: %w", path, err)
	}

	// --- Pipeline: select -> filter -> groupby/agg -> sort ---

	// Select columns.
	if readSelect != "" {
		cols := strings.Split(readSelect, ",")
		for i := range cols {
			cols[i] = strings.TrimSpace(cols[i])
		}
		df, err = df.Select(cols...)
		if err != nil {
			return fmt.Errorf("select: %w", err)
		}
	}

	// Filter rows.
	if readFilter != "" {
		df, err = df.Query(readFilter)
		if err != nil {
			return fmt.Errorf("filter: %w", err)
		}
	}

	// GroupBy + Aggregation.
	if readGroupBy != "" {
		gb := df.GroupBy(readGroupBy)
		df, err = applyAgg(gb, readAgg)
		if err != nil {
			return err
		}
	}

	// Sort.
	if readSort != "" {
		ascending := !readSortDesc
		df, err = df.SortBy([]string{readSort}, []bool{ascending})
		if err != nil {
			return fmt.Errorf("sort: %w", err)
		}
	}

	// --- Output ---

	// Write to file if --output set.
	if readOutput != "" {
		outFormat := readFormat
		if outFormat == "" {
			outFormat, err = inferFormat(readOutput)
			if err != nil {
				return err
			}
		}
		return writeFile(df, readOutput, outFormat)
	}

	// Display mode: shape, dtypes, describe, head, tail.
	if readShape {
		rows, cols := df.Shape()
		fmt.Printf("(%d, %d)\n", rows, cols)
		return nil
	}

	if readDTypes {
		dtypes := df.DTypes()
		for _, col := range df.Columns() {
			fmt.Printf("%s: %s\n", col, dtypes[col])
		}
		return nil
	}

	if readDescribe {
		desc := df.Describe()
		fmt.Println(desc.String())
		return nil
	}

	if readTail > 0 {
		fmt.Println(df.Tail(readTail).String())
		return nil
	}

	// Default: head (5 if not specified).
	n := readHead
	if n <= 0 {
		n = 5
	}
	rows, _ := df.Shape()
	if n > rows {
		n = rows
	}
	fmt.Println(df.Head(n).String())
	return nil
}

// applyAgg applies an aggregation function to a GroupBy.
func applyAgg(gb interface{ Sum() dataframe.DataFrame; Mean() dataframe.DataFrame; Count() dataframe.DataFrame; Min() dataframe.DataFrame; Max() dataframe.DataFrame }, agg string) (dataframe.DataFrame, error) {
	switch strings.ToLower(agg) {
	case "sum":
		return gb.Sum(), nil
	case "mean":
		return gb.Mean(), nil
	case "count":
		return gb.Count(), nil
	case "min":
		return gb.Min(), nil
	case "max":
		return gb.Max(), nil
	case "":
		return gb.Count(), nil // default aggregation
	default:
		return dataframe.DataFrame{}, fmt.Errorf("unsupported aggregation: %s (use sum, mean, count, min, max)", agg)
	}
}

package main

import (
	"bytes"
	"os"
	"strings"
	"testing"

	"github.com/spf13/cobra"
)

// executeCommand runs the root command with the given args and returns stdout output.
func executeCommand(args ...string) (string, error) {
	buf := new(bytes.Buffer)
	rootCmd.SetOut(buf)
	rootCmd.SetErr(buf)
	rootCmd.SetArgs(args)

	err := rootCmd.Execute()
	return buf.String(), err
}

// resetFlags resets all read flags to defaults between tests.
func resetFlags() {
	readHead = 0
	readTail = 0
	readDescribe = false
	readShape = false
	readDTypes = false
	readSelect = ""
	readFilter = ""
	readGroupBy = ""
	readAgg = ""
	readSort = ""
	readSortDesc = false
	readOutput = ""
	readFormat = ""
}
```

**Note:** Because the CLI uses package-level variables, we reset them between tests. An alternative is to use `cmd.ResetFlags()` but explicit reset is simpler here.

```go
package main

import (
	"fmt"
	"strings"

	"github.com/spf13/cobra"
)

var (
	convertFrom   string
	convertTo     string
	convertSelect string
)

var convertCmd = &cobra.Command{
	Use:   "convert <input> <output>",
	Short: "Convert between data file formats",
	Long:  "Convert a data file from one format to another (CSV, JSON, Parquet). Format is inferred from file extensions unless --from/--to are specified.",
	Args:  cobra.ExactArgs(2),
	RunE:  runConvert,
}

func init() {
	convertCmd.Flags().StringVar(&convertFrom, "from", "", "Input format: csv, json, parquet (default: infer from extension)")
	convertCmd.Flags().StringVar(&convertTo, "to", "", "Output format: csv, json, parquet (default: infer from extension)")
	convertCmd.Flags().StringVar(&convertSelect, "select", "", "Select columns (comma-separated)")

	rootCmd.AddCommand(convertCmd)
}

func runConvert(cmd *cobra.Command, args []string) error {
	inputPath := args[0]
	outputPath := args[1]

	// Determine input format.
	inFormat := convertFrom
	if inFormat == "" {
		var err error
		inFormat, err = inferFormat(inputPath)
		if err != nil {
			return fmt.Errorf("cannot infer input format: %w (use --from to specify)", err)
		}
	}

	// Determine output format.
	outFormat := convertTo
	if outFormat == "" {
		var err error
		outFormat, err = inferFormat(outputPath)
		if err != nil {
			return fmt.Errorf("cannot infer output format: %w (use --to to specify)", err)
		}
	}

	// Load input file.
	df, err := loadFile(inputPath, inFormat)
	if err != nil {
		return fmt.Errorf("failed to read %s: %w", inputPath, err)
	}

	// Select columns if specified.
	if convertSelect != "" {
		cols := strings.Split(convertSelect, ",")
		for i := range cols {
			cols[i] = strings.TrimSpace(cols[i])
		}
		df, err = df.Select(cols...)
		if err != nil {
			return fmt.Errorf("select: %w", err)
		}
	}

	// Write output file.
	if err := writeFile(df, outputPath, outFormat); err != nil {
		return fmt.Errorf("failed to write %s: %w", outputPath, err)
	}

	rows, cols := df.Shape()
	fmt.Fprintf(cmd.OutOrStdout(), "Converted %s (%s) -> %s (%s) [%d rows, %d cols]\n",
		inputPath, inFormat, outputPath, outFormat, rows, cols)
	return nil
}
```
