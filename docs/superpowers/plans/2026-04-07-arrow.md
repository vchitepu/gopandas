> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build an internal adapter package wrapping apache/arrow-go to provide typed array construction, value extraction, slicing, and null handling for gopandas.

**Architecture:** Package `arrowutil/` provides builder functions for each supported dtype, a generic value extractor, zero-copy slicing, and null bitmap access. All other gopandas packages use this instead of raw Arrow APIs.

**Tech Stack:** Go 1.21+, github.com/apache/arrow-go/v18

```go
package arrowutil

import (
	"time"

	"github.com/apache/arrow-go/v18/arrow"
	"github.com/apache/arrow-go/v18/arrow/array"
	"github.com/apache/arrow-go/v18/arrow/memory"
)

// BuildInt64Array creates an Arrow Int64 array from a Go slice.
// The caller is responsible for calling Release() on the returned array.
func BuildInt64Array(alloc memory.Allocator, values []int64) *array.Int64 {
	bldr := array.NewInt64Builder(alloc)
	defer bldr.Release()
	bldr.AppendValues(values, nil)
	return bldr.NewInt64Array()
}
```

```go
// BuildTimestampArray creates an Arrow Timestamp (microsecond, UTC) array from a Go time.Time slice.
// Values are stored as microseconds since Unix epoch.
func BuildTimestampArray(alloc memory.Allocator, values []time.Time) *array.Timestamp {
	dt := &arrow.TimestampType{Unit: arrow.Microsecond, TimeZone: "UTC"}
	bldr := array.NewTimestampBuilder(alloc, dt)
	defer bldr.Release()
	for _, v := range values {
		bldr.Append(arrow.Timestamp(v.UnixMicro()))
	}
	return bldr.NewTimestampArray()
}
```

```go
// GetValue extracts a single value from an Arrow array at position i.
// Returns nil for null values. Returns an error for out-of-bounds access
// or unsupported array types.
func GetValue(arr arrow.Array, i int) (any, error) {
	if i < 0 || i >= arr.Len() {
		return nil, fmt.Errorf("arrowutil.GetValue: index %d out of bounds [0, %d)", i, arr.Len())
	}
	if arr.IsNull(i) {
		return nil, nil
	}
	switch a := arr.(type) {
	case *array.Int64:
		return a.Value(i), nil
	case *array.Float64:
		return a.Value(i), nil
	case *array.String:
		return a.Value(i), nil
	case *array.Boolean:
		return a.Value(i), nil
	case *array.Timestamp:
		us := int64(a.Value(i))
		return time.Unix(0, us*int64(time.Microsecond)).UTC(), nil
	default:
		return nil, fmt.Errorf("arrowutil.GetValue: unsupported array type %T", arr)
	}
}
```

```go
// BuildArray creates an Arrow array from a generic []any slice using the specified DType.
// nil values in the input slice are stored as Arrow nulls.
// Returns an error for unsupported dtypes or type assertion failures.
func BuildArray(alloc memory.Allocator, values []any, dt dtype.DType) (arrow.Array, error) {
	switch dt {
	case dtype.Int64:
		return buildInt64FromAny(alloc, values)
	case dtype.Float64:
		return buildFloat64FromAny(alloc, values)
	case dtype.String:
		return buildStringFromAny(alloc, values)
	case dtype.Bool:
		return buildBoolFromAny(alloc, values)
	case dtype.Timestamp:
		return buildTimestampFromAny(alloc, values)
	default:
		return nil, fmt.Errorf("arrowutil.BuildArray: unsupported dtype %v", dt)
	}
}

func buildInt64FromAny(alloc memory.Allocator, values []any) (arrow.Array, error) {
	bldr := array.NewInt64Builder(alloc)
	defer bldr.Release()
	for i, v := range values {
		if v == nil {
			bldr.AppendNull()
			continue
		}

			return nil, fmt.Errorf("arrowutil.BuildArray: index %d: expected int64, got %T", i, v)
```

```go
// GetValue extracts a single value from an Arrow array at position i.
// Returns nil for null values. Returns an error for out-of-bounds access
// or unsupported array types.
func GetValue(arr arrow.Array, i int) (any, error) {
	if i < 0 || i >= arr.Len() {
		return nil, fmt.Errorf("arrowutil.GetValue: index %d out of bounds [0, %d)", i, arr.Len())
	}
	if arr.IsNull(i) {
		return nil, nil
	}
	switch a := arr.(type) {
	case *array.Int64:
		return a.Value(i), nil
	case *array.Float64:
		return a.Value(i), nil
	case *array.String:
		return a.Value(i), nil
	case *array.Boolean:
		return a.Value(i), nil
	case *array.Timestamp:
		us := int64(a.Value(i))
		return time.Unix(0, us*int64(time.Microsecond)).UTC(), nil
	default:
		return nil, fmt.Errorf("arrowutil.GetValue: unsupported array type %T", arr)
	}
}
```
