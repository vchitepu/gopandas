```go
package groupby

import (
	"fmt"
	"strings"

	"github.com/vchitepu/gopandas/dataframe"
)

// GroupBy holds the result of a DataFrame.GroupBy() operation.
// It stores the source DataFrame, the group key column names,
// and a pre-computed map of composite key -> row indices.
type GroupBy struct {
	df     dataframe.DataFrame
	keys   []string
	groups map[string][]int // composite key "val1|val2" -> row indices
}

// NewGroupBy creates a GroupBy from a DataFrame and one or more key column names.
// It immediately computes the group assignments by iterating through all rows once.
func NewGroupBy(df dataframe.DataFrame, keys ...string) GroupBy {
	gb := GroupBy{
		df:   df,
		keys: keys,
	}
	gb.buildGroups()
	return gb
}

// buildGroups iterates through the DataFrame once and populates the groups map.
// Multi-column keys are serialized as "val1|val2|..." strings.
func (gb *GroupBy) buildGroups() {
	gb.groups = make(map[string][]int)
	nRows := gb.df.Len()

	for i := 0; i < nRows; i++ {
		key := gb.compositeKey(i)
		gb.groups[key] = append(gb.groups[key], i)
	}
}

// compositeKey builds the composite group key string for row i.
// For a single key column, it returns the string representation of the value.
// For multiple key columns, values are joined with "|".
func (gb *GroupBy) compositeKey(row int) string {
	parts := make([]string, len(gb.keys))
	for j, col := range gb.keys {
		val, err := gb.df.At(row, col)
		if err != nil {
			parts[j] = "<nil>"
		} else {
			parts[j] = fmt.Sprintf("%v", val)
		}
	}
	return strings.Join(parts, "|")
}
```

```go
// Groups returns a copy of the internal group map.
// Keys are composite key strings (e.g. "val1|val2"), values are row index slices.
func (gb GroupBy) Groups() map[string][]int {
	result := make(map[string][]int, len(gb.groups))
	for k, v := range gb.groups {
		cp := make([]int, len(v))
		copy(cp, v)
		result[k] = cp
	}
	return result
}
```

```go
// sortedGroupKeys returns the composite group keys in sorted order
// for deterministic output.
func (gb GroupBy) sortedGroupKeys() []string {
	keys := make([]string, 0, len(gb.groups))
	for k := range gb.groups {
		keys = append(keys, k)
	}
	sort.Strings(keys)
	return keys
}

// Size returns a Series[int64] with the number of rows in each group.
// The Series is indexed by the composite group key strings.
func (gb GroupBy) Size() series.Series[int64] {
	keys := gb.sortedGroupKeys()
	labels := make([]string, len(keys))
	values := make([]int64, len(keys))
	for i, k := range keys {
		labels[i] = k
		values[i] = int64(len(gb.groups[k]))
	}
	idx := index.NewStringIndex(labels, "")
	return series.New[int64](memory.DefaultAllocator, values, idx, "size")
}
```

```go
// subDF creates a new DataFrame containing only the rows at the given positions.
// The result has a fresh RangeIndex and retains all columns from the source DataFrame.
func (gb GroupBy) subDF(positions []int) (dataframe.DataFrame, error) {
	nRows := len(positions)
	cols := gb.df.Columns()

	colData := make(map[string]any, len(cols))
	for _, col := range cols {
		vals := make([]any, nRows)
		for i, pos := range positions {
			val, err := gb.df.At(pos, col)
			if err != nil {
				return dataframe.DataFrame{}, fmt.Errorf("groupby.subDF: %w", err)
			}
			vals[i] = val
		}
		colData[col] = vals
	}

	return dataframe.FromRecords(buildRecords(cols, colData, nRows))
}

// buildRecords converts columnar data into the []map[string]any format
// required by dataframe.FromRecords.
func buildRecords(cols []string, colData map[string]any, nRows int) []map[string]any {
	records := make([]map[string]any, nRows)
	for i := 0; i < nRows; i++ {
		rec := make(map[string]any, len(cols))
		for _, col := range cols {
			vals := colData[col].([]any)
			rec[col] = vals[i]
		}
		records[i] = rec
	}
	return records
}
```

```go
func (gb GroupBy) valueColumns() []string {
	keySet := make(map[string]bool, len(gb.keys))
	for _, k := range gb.keys {
		keySet[k] = true

	}

// numericValueColumns returns non-key columns that have numeric dtype (Int64 or Float64).
func (gb GroupBy) numericValueColumns() []string {
	keySet := make(map[string]bool, len(gb.keys))
	for _, k := range gb.keys {
		keySet[k] = true
	}
	var result []string
	for _, col := range gb.df.Columns() {
		if keySet[col] {

		}

		if err != nil {
			continue
		}
		dt := s.DType()
		if dt == dtype.Int64 || dt == dtype.Float64 {

		}

// extractFloat64Values extracts float64 values from the source DataFrame column
// at the given row positions. Non-numeric values are skipped (returned slice may be shorter).
func (gb GroupBy) extractFloat64Values(col string, positions []int) []float64 {
	result := make([]float64, 0, len(positions))
	for _, pos := range positions {
		val, err := gb.df.At(pos, col)
		if err != nil || val == nil {
			continue
		}
		switch v := val.(type) {
		case int64:
			result = append(result, float64(v))
		case float64:
			result = append(result, v)
		}
	}
	return result
}

// aggregateNumeric applies fn to each numeric value column for each group,
// producing a result DataFrame with group keys as the index (StringIndex).
// Each row is one group; columns are the numeric non-key columns.
func (gb GroupBy) aggregateNumeric(fn aggFunc) (dataframe.DataFrame, error) {

	valueCols := gb.numericValueColumns()

	if len(valueCols) == 0 {
		return dataframe.DataFrame{}, fmt.Errorf("groupby: no numeric columns to aggregate")
	}

	// Build result columns
	resultData := make(map[string]any, len(valueCols))
	for _, col := range valueCols {

			vals[i] = fn(floats)

		return dataframe.DataFrame{}, fmt.Errorf("groupby.aggregateNumeric: %w", err)
```

```go
// aggregateNumeric applies fn to each numeric value column for each group,
// producing a result DataFrame. The group key columns are included in the result.
func (gb GroupBy) aggregateNumeric(fn aggFunc) (dataframe.DataFrame, error) {
	groupKeys := gb.sortedGroupKeys()
	valueCols := gb.numericValueColumns()

	if len(valueCols) == 0 {
		return dataframe.DataFrame{}, fmt.Errorf("groupby: no numeric columns to aggregate")
	}

	// Build result columns: start with key columns, then value columns
	resultData := make(map[string]any)

	// Parse composite keys back into individual key column values
	for ki, keyCol := range gb.keys {
		keyVals := make([]string, len(groupKeys))
		for i, gk := range groupKeys {
			parts := strings.Split(gk, "|")
			if ki < len(parts) {
				keyVals[i] = parts[ki]
			}
		}
		resultData[keyCol] = keyVals
	}

	// Aggregate each numeric value column
	for _, col := range valueCols {
		vals := make([]float64, len(groupKeys))
		for i, gk := range groupKeys {
			positions := gb.groups[gk]
			floats := gb.extractFloat64Values(col, positions)
			vals[i] = fn(floats)
		}
		resultData[col] = vals
	}

	return dataframe.New(resultData)
}
```

```go
// Count returns a DataFrame with the count of non-null values in each column per group.
// Unlike other aggregations, Count includes all non-key columns (not just numeric).
func (gb GroupBy) Count() (dataframe.DataFrame, error) {
	groupKeys := gb.sortedGroupKeys()
	valueCols := gb.valueColumns()

	if len(valueCols) == 0 {
		return dataframe.DataFrame{}, fmt.Errorf("groupby: no value columns to count")
	}

	resultData := make(map[string]any)

	// Parse composite keys back into individual key column values
	for ki, keyCol := range gb.keys {
		keyVals := make([]string, len(groupKeys))
		for i, gk := range groupKeys {
			parts := strings.Split(gk, "|")
			if ki < len(parts) {
				keyVals[i] = parts[ki]
			}
		}
		resultData[keyCol] = keyVals
	}

	// Count non-null values per group per column
	for _, col := range valueCols {
		counts := make([]int64, len(groupKeys))
		for i, gk := range groupKeys {
			positions := gb.groups[gk]
			var cnt int64
			for _, pos := range positions {
				val, err := gb.df.At(pos, col)
				if err == nil && val != nil {
					cnt++
				}
			}
			counts[i] = cnt
		}
		resultData[col] = counts
	}

	return dataframe.New(resultData)
}
```

```go
// Std returns a DataFrame with the sample standard deviation (ddof=1) of each
// numeric column per group. Groups with fewer than 2 values produce NaN.
func (gb GroupBy) Std() (dataframe.DataFrame, error) {
	return gb.aggregateNumeric(func(vals []float64) float64 {
		n := len(vals)
		if n < 2 {
			return math.NaN()
		}
		// Compute mean
		var sum float64
		for _, v := range vals {
			sum += v
		}
		mean := sum / float64(n)

		// Compute sample variance (ddof=1)
		var sqDiffSum float64
		for _, v := range vals {
			diff := v - mean
			sqDiffSum += diff * diff
		}
		variance := sqDiffSum / float64(n-1)
		return math.Sqrt(variance)
	})
}
```

```go
// First returns a DataFrame with the first row from each group.
// All columns (not just numeric) are included in the result.
func (gb GroupBy) First() (dataframe.DataFrame, error) {
	return gb.aggregateFirstLast(true)
}

// aggregateFirstLast is the shared implementation for First() and Last().
// If first is true, takes the first row index from each group; otherwise the last.
func (gb GroupBy) aggregateFirstLast(first bool) (dataframe.DataFrame, error) {
	groupKeys := gb.sortedGroupKeys()
	allCols := gb.df.Columns()

	resultData := make(map[string]any)

	// Initialize typed slices for each column based on the source DataFrame's dtype
	for _, col := range allCols {
		s, err := gb.df.Col(col)
		if err != nil {
			return dataframe.DataFrame{}, fmt.Errorf("groupby.First/Last: %w", err)
		}
		switch s.DType() {
		case dtype.String:
			resultData[col] = make([]string, len(groupKeys))
		case dtype.Int64:
			resultData[col] = make([]int64, len(groupKeys))
		case dtype.Float64:
			resultData[col] = make([]float64, len(groupKeys))
		case dtype.Bool:
			resultData[col] = make([]bool, len(groupKeys))
		default:
			resultData[col] = make([]string, len(groupKeys))
		}
	}

	for i, gk := range groupKeys {
		positions := gb.groups[gk]
		var pos int
		if first {
			pos = positions[0]
		} else {
			pos = positions[len(positions)-1]
		}

		for _, col := range allCols {
			val, err := gb.df.At(pos, col)
			if err != nil {
				return dataframe.DataFrame{}, fmt.Errorf("groupby.First/Last: %w", err)
			}
			switch slice := resultData[col].(type) {
			case []string:
				if v, ok := val.(string); ok {
					slice[i] = v
				} else {
					slice[i] = fmt.Sprintf("%v", val)
				}
			case []int64:
				if v, ok := val.(int64); ok {
					slice[i] = v
				}
			case []float64:
				if v, ok := val.(float64); ok {
					slice[i] = v
				}
			case []bool:
				if v, ok := val.(bool); ok {
					slice[i] = v
				}
			}
		}
	}

	return dataframe.New(resultData)
}
```

```go
package groupby

import (
	"fmt"
	"math"
	"strings"

	"github.com/vchitepu/gopandas/dataframe"

// Agg applies a named aggregation function to each specified column per group.
// The fns map keys are column names, values are function names:
// "sum", "mean", "count", "min", "max", "std", "first", "last".
// All result columns are float64 for consistency.
// Returns an error if any column is not found or the function name is unknown.
func (gb GroupBy) Agg(fns map[string]string) (dataframe.DataFrame, error) {
	groupKeys := gb.sortedGroupKeys()

	resultData := make(map[string]any)

	// Add key columns
	for ki, keyCol := range gb.keys {
		keyVals := make([]string, len(groupKeys))
		for i, gk := range groupKeys {
			parts := strings.Split(gk, "|")
			if ki < len(parts) {
				keyVals[i] = parts[ki]
			}
		}
		resultData[keyCol] = keyVals
	}

	// Apply each function to its column
	for col, fnName := range fns {
		fn, err := resolveAggFunc(fnName)
		if err != nil {
			return dataframe.DataFrame{}, fmt.Errorf("groupby.Agg: column %q: %w", col, err)
		}

		vals := make([]float64, len(groupKeys))
		for i, gk := range groupKeys {
			positions := gb.groups[gk]
			vals[i] = fn(gb, col, positions)
		}
		resultData[col] = vals
	}

	return dataframe.New(resultData)
}

// aggColFunc is a function that aggregates a single column for a set of row positions.
type aggColFunc func(gb GroupBy, col string, positions []int) float64

// resolveAggFunc maps a string function name to the corresponding aggColFunc.
func resolveAggFunc(name string) (aggColFunc, error) {
	switch name {
	case "sum":
		return aggSum, nil
	case "mean":
		return aggMean, nil
	case "count":
		return aggCount, nil
	case "min":
		return aggMin, nil
	case "max":
		return aggMax, nil
	case "std":
		return aggStd, nil
	case "first":
		return aggFirst, nil
	case "last":
		return aggLast, nil
	default:
		return nil, fmt.Errorf("unknown aggregation function %q (supported: sum, mean, count, min, max, std, first, last)", name)
	}
}

func aggSum(gb GroupBy, col string, positions []int) float64 {
	vals := gb.extractFloat64Values(col, positions)
	var sum float64
	for _, v := range vals {
		sum += v
	}
	return sum
}

func aggMean(gb GroupBy, col string, positions []int) float64 {
	vals := gb.extractFloat64Values(col, positions)
	if len(vals) == 0 {
		return math.NaN()
	}
	var sum float64
	for _, v := range vals {
		sum += v
	}
	return sum / float64(len(vals))
}

func aggCount(gb GroupBy, col string, positions []int) float64 {
	var cnt float64
	for _, pos := range positions {
		val, err := gb.df.At(pos, col)
		if err == nil && val != nil {
			cnt++
		}
	}
	return cnt
}

func aggMin(gb GroupBy, col string, positions []int) float64 {
	vals := gb.extractFloat64Values(col, positions)
	if len(vals) == 0 {
		return math.NaN()
	}
	min := vals[0]
	for _, v := range vals[1:] {
		if v < min {
			min = v
		}
	}
	return min
}

func aggMax(gb GroupBy, col string, positions []int) float64 {
	vals := gb.extractFloat64Values(col, positions)
	if len(vals) == 0 {
		return math.NaN()
	}
	max := vals[0]
	for _, v := range vals[1:] {
		if v > max {
			max = v
		}
	}
	return max
}

func aggStd(gb GroupBy, col string, positions []int) float64 {
	vals := gb.extractFloat64Values(col, positions)
	n := len(vals)
	if n < 2 {
		return math.NaN()
	}
	var sum float64
	for _, v := range vals {
		sum += v
	}
	mean := sum / float64(n)
	var sqDiffSum float64
	for _, v := range vals {
		diff := v - mean
		sqDiffSum += diff * diff
	}
	return math.Sqrt(sqDiffSum / float64(n-1))
}

func aggFirst(gb GroupBy, col string, positions []int) float64 {
	if len(positions) == 0 {
		return math.NaN()
	}
	vals := gb.extractFloat64Values(col, positions[:1])
	if len(vals) == 0 {
		// Non-numeric column, count as 0
		return 0
	}
	return vals[0]
}

func aggLast(gb GroupBy, col string, positions []int) float64 {
	if len(positions) == 0 {
		return math.NaN()
	}
	vals := gb.extractFloat64Values(col, positions[len(positions)-1:])
	if len(vals) == 0 {
		return 0
	}
	return vals[0]
}
```

```go
// Apply calls fn once per group with the group's sub-DataFrame.
// The results are concatenated vertically into a single DataFrame.
// fn must return DataFrames with consistent column schemas.
// Returns an error if fn returns an error for any group.
func (gb GroupBy) Apply(fn func(dataframe.DataFrame) (dataframe.DataFrame, error)) (dataframe.DataFrame, error) {
	groupKeys := gb.sortedGroupKeys()

	var results []dataframe.DataFrame
	for _, gk := range groupKeys {
		positions := gb.groups[gk]
		sub, err := gb.subDF(positions)
		if err != nil {
			return dataframe.DataFrame{}, fmt.Errorf("groupby.Apply: building sub-DataFrame for group %q: %w", gk, err)
		}
		result, err := fn(sub)
		if err != nil {
			return dataframe.DataFrame{}, fmt.Errorf("groupby.Apply: fn error for group %q: %w", gk, err)
		}
		results = append(results, result)
	}

	if len(results) == 0 {
		return dataframe.DataFrame{}, nil
	}

	// Concatenate all result DataFrames vertically
	return concatDataFrames(results)
}

// concatDataFrames concatenates multiple DataFrames vertically.
// All DataFrames must have the same columns.
func concatDataFrames(dfs []dataframe.DataFrame) (dataframe.DataFrame, error) {
	if len(dfs) == 0 {
		return dataframe.DataFrame{}, nil
	}
	if len(dfs) == 1 {
		return dfs[0], nil
	}

	cols := dfs[0].Columns()
	totalRows := 0
	for _, df := range dfs {
		totalRows += df.Len()
	}

	// Collect all records
	records := make([]map[string]any, 0, totalRows)
	for _, df := range dfs {
		for i := 0; i < df.Len(); i++ {
			rec := make(map[string]any, len(cols))
			for _, col := range cols {
				val, err := df.At(i, col)
				if err != nil {
					// Column not in this DataFrame — skip (nil)
					continue
				}
				rec[col] = val
			}
			records = append(records, rec)
		}
	}

	return dataframe.FromRecords(records)
}
```

```go
// TransformFunc is a function that receives a column name, row positions for a group,
// and the source DataFrame. It must return a float64 slice with one value per position
// (same order as positions).
type TransformFunc func(col string, positions []int, src dataframe.DataFrame) ([]float64, error)

// Transform applies fn to each numeric value column for each group, producing a
// result DataFrame with the same row count and order as the source DataFrame.
// Each group's transformed values are placed back into their original row positions.
// The result includes the key columns and the transformed numeric columns.
func (gb GroupBy) Transform(fn TransformFunc) (dataframe.DataFrame, error) {
	numericCols := gb.numericValueColumns()
	if len(numericCols) == 0 {
		return dataframe.DataFrame{}, fmt.Errorf("groupby.Transform: no numeric columns to transform")
	}

	nRows := gb.df.Len()

	// Initialize result arrays — one per numeric column
	colResults := make(map[string][]float64, len(numericCols))
	for _, col := range numericCols {
		colResults[col] = make([]float64, nRows)
	}

	// Apply transform per group per column
	for _, gk := range gb.sortedGroupKeys() {
		positions := gb.groups[gk]
		for _, col := range numericCols {
			transformed, err := fn(col, positions, gb.df)
			if err != nil {
				return dataframe.DataFrame{}, fmt.Errorf("groupby.Transform: group %q col %q: %w", gk, col, err)
			}
			if len(transformed) != len(positions) {
				return dataframe.DataFrame{}, fmt.Errorf(
					"groupby.Transform: group %q col %q: fn returned %d values, want %d",
					gk, col, len(transformed), len(positions),
				)
			}
			for i, pos := range positions {
				colResults[col][pos] = transformed[i]
			}
		}
	}

	// Build result DataFrame with key columns + transformed columns
	resultData := make(map[string]any)

	// Copy key columns from source
	for _, keyCol := range gb.keys {
		vals := make([]string, nRows)
		for i := 0; i < nRows; i++ {
			v, _ := gb.df.At(i, keyCol)
			vals[i] = fmt.Sprintf("%v", v)
		}
		resultData[keyCol] = vals
	}

	// Add transformed numeric columns
	for _, col := range numericCols {
		resultData[col] = colResults[col]
	}

	return dataframe.New(resultData)
}
```
