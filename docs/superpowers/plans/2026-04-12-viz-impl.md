**Goal:** Add `--viz` flag to the `gopandas read` CLI with styled tables, bar/histogram/line charts, and a summary dashboard — all rendered in-terminal using lipgloss.

**Architecture:** New `cmd/gopandas/viz/` package with five files: `theme.go` (lipgloss theme + auto-detect), `table.go` (styled DataFrame table), `chart.go` (bar, histogram, line renderers), `summary.go` (full dashboard panel), `render.go` (dispatch entrypoint + VizOptions). The `read` command gains `--viz`, `--x`, `--y`, `--bins`, `--theme` flags and delegates to `viz.Render()` after the existing pipeline runs.

**Tech Stack:** Go 1.25, charmbracelet/lipgloss v1.1.0, golang.org/x/term, spf13/cobra, existing gopandas lib (dataframe, series, dtype packages).

package viz

import (
	"os"
	"strconv"
	"strings"

	"github.com/charmbracelet/lipgloss"
)

// Theme holds all lipgloss styles used by the viz renderers.
type Theme struct {
	// Text
	Normal lipgloss.Style
	Bold   lipgloss.Style
	Dim    lipgloss.Style

	// Headings (section labels in dashboard)
	SectionHeader lipgloss.Style

	// Table
	TableHeader lipgloss.Style
	TableBorder lipgloss.Color

	// Charts
	BarColor    lipgloss.Color
	AxisColor   lipgloss.Style
	ChartBorder lipgloss.Color

	// Dashboard outer panel
	PanelBorder lipgloss.Color
	PanelTitle  lipgloss.Style
}

// DarkTheme returns the built-in dark theme (matching weave palette).
func DarkTheme() Theme {
	return Theme{
		Normal: lipgloss.NewStyle(),
		Bold:   lipgloss.NewStyle().Bold(true),
		Dim:    lipgloss.NewStyle().Faint(true),

		SectionHeader: lipgloss.NewStyle().Bold(true).Foreground(lipgloss.Color("#C9A86A")),

		TableHeader: lipgloss.NewStyle().Bold(true).Foreground(lipgloss.Color("#8BA4D4")),
		TableBorder: lipgloss.Color("#3A3F4B"),

		BarColor:    lipgloss.Color("#8BA4D4"),
		AxisColor:   lipgloss.NewStyle().Faint(true),
		ChartBorder: lipgloss.Color("#3A3F4B"),

		PanelBorder: lipgloss.Color("#7B74A6"),
		PanelTitle:  lipgloss.NewStyle().Bold(true),
	}
}

// LightTheme returns the built-in light theme.
func LightTheme() Theme {
	return Theme{
		Normal: lipgloss.NewStyle(),
		Bold:   lipgloss.NewStyle().Bold(true),
		Dim:    lipgloss.NewStyle().Faint(true),

		SectionHeader: lipgloss.NewStyle().Bold(true).Foreground(lipgloss.Color("#8D6B3F")),

		TableHeader: lipgloss.NewStyle().Bold(true).Foreground(lipgloss.Color("#3F5F8A")),
		TableBorder: lipgloss.Color("#C2C7D0"),

		BarColor:    lipgloss.Color("#3F5F8A"),
		AxisColor:   lipgloss.NewStyle().Faint(true),
		ChartBorder: lipgloss.Color("#C2C7D0"),

		PanelBorder: lipgloss.Color("#7D73A3"),
		PanelTitle:  lipgloss.NewStyle().Bold(true),
	}
}

// Detect returns a Theme based on the flag override, env vars, or heuristics.
// flagValue is the --theme flag value ("dark", "light", or "" for auto).
func Detect(flagValue string) Theme {
	mode := detectMode(flagValue)
	if mode == "light" {
		return LightTheme()
	}
	return DarkTheme()
}

func detectMode(flagValue string) string {
	// 1. Flag override
	if flagValue == "dark" || flagValue == "light" {
		return flagValue
	}

	// 2. GOPANDAS_THEME env var
	if env := os.Getenv("GOPANDAS_THEME"); env == "dark" || env == "light" {
		return env
	}

	// 3. COLORFGBG — format "fg;bg", bg < 128 means dark
	if colorfgbg := os.Getenv("COLORFGBG"); colorfgbg != "" {
		parts := strings.Split(colorfgbg, ";")
		if len(parts) >= 2 {
			if bg, err := strconv.Atoi(parts[len(parts)-1]); err == nil {
				if bg < 128 {
					return "dark"
				}
				return "light"
			}
		}
	}

	// 4. TERM_PROGRAM heuristics
	if os.Getenv("TERM_PROGRAM") == "Apple_Terminal" {
		return "light"
	}

	// 5. Fallback
	return "dark"
}
```

package viz

import (
	"fmt"
	"strings"

	"github.com/vchitepu/gopandas/lib/dataframe"
	"github.com/vchitepu/gopandas/lib/dtype"
)

// VizOptions configures which visualization to render.
type VizOptions struct {
	Type      string // "bar" | "histogram" | "line" | "table" | "summary"
	XCol      string
	YCol      string
	Bins      int
	ThemeMode string // "dark" | "light" | "" (auto)
	Filename  string // for dashboard title
}

// ChartOptions is passed to individual chart renderers.
type ChartOptions struct {
	Type  string // "bar" | "histogram" | "line"
	XCol  string
	YCol  string
	Bins  int
	Title string
}

// validVizTypes lists all supported --viz values.
var validVizTypes = []string{"bar", "histogram", "line", "table", "summary"}

// Render dispatches to the correct renderer based on opts.Type.
// Returns plain df.String() if opts.Type is empty or isTTY is false.
func Render(df dataframe.DataFrame, opts VizOptions, termWidth int, isTTY bool) (string, error) {
	// No viz requested — plain output
	if opts.Type == "" {
		return df.String(), nil
	}

	// Non-TTY fallback
	if !isTTY {
		return df.String(), nil
	}

	// Validate viz type
	if !isValidVizType(opts.Type) {
		return "", fmt.Errorf("unknown viz type %q (supported: %s)", opts.Type, strings.Join(validVizTypes, ", "))
	}

	th := Detect(opts.ThemeMode)

	switch opts.Type {
	case "table":
		return RenderTable(df, th, termWidth), nil

	case "summary":
		return RenderSummary(df, opts.Filename, th, termWidth), nil

	case "bar", "histogram", "line":
		chartOpts, err := buildChartOptions(df, opts)
		if err != nil {
			return "", err
		}
		switch opts.Type {
		case "bar":
			return RenderBar(df, chartOpts, th, termWidth), nil
		case "histogram":
			return RenderHistogram(df, chartOpts, th, termWidth), nil
		case "line":
			return RenderLine(df, chartOpts, th, termWidth), nil
		}
	}

	return df.String(), nil
}

// buildChartOptions resolves x/y columns and constructs ChartOptions.
func buildChartOptions(df dataframe.DataFrame, opts VizOptions) (ChartOptions, error) {
	requireNumericX := opts.Type == "histogram"
	requireNumericY := opts.Type == "bar" || opts.Type == "line"

	xCol, err := resolveColumn(df, opts.XCol, "x", requireNumericX)
	if err != nil {
		return ChartOptions{}, err
	}

	var yCol string
	if opts.Type != "histogram" {
		yCol, err = resolveColumn(df, opts.YCol, "y", requireNumericY)
		if err != nil {
			return ChartOptions{}, err
		}
	}

	bins := opts.Bins
	if bins <= 0 {
		bins = 10
	}

	title := opts.Filename
	if opts.Type == "histogram" {
		title += " | distribution of " + xCol
	} else {
		title += " | " + xCol + " vs " + yCol
	}

	return ChartOptions{
		Type:  opts.Type,
		XCol:  xCol,
		YCol:  yCol,
		Bins:  bins,
		Title: title,
	}, nil
}

// resolveColumn resolves a column name: uses the explicit value if given,
// otherwise picks a default. If requireNumeric is true, the resolved column
// must be Int64 or Float64.
func resolveColumn(df dataframe.DataFrame, explicit string, axis string, requireNumeric bool) (string, error) {
	columns := df.Columns()
	dtypes := df.DTypes()

	if explicit != "" {
		// Check it exists
		found := false
		for _, c := range columns {
			if c == explicit {
				found = true
				break
			}
		}
		if !found {
			return "", fmt.Errorf("--%s column %q not found (available: %s)", axis, explicit, strings.Join(columns, ", "))
		}
		// Check numeric if required
		if requireNumeric {
			d := dtypes[explicit]
			if d != dtype.Int64 && d != dtype.Float64 {
				return "", fmt.Errorf("--%s column %q is not numeric (dtype: %s)", axis, explicit, d)
			}
		}
		return explicit, nil
	}

	// Default: first column (x) or first numeric column (y)
	if requireNumeric {
		for _, c := range columns {
			d := dtypes[c]
			if d == dtype.Int64 || d == dtype.Float64 {
				return c, nil
			}
		}
		return "", fmt.Errorf("no numeric column found for --%s default (available: %s)", axis, strings.Join(columns, ", "))
	}

	if len(columns) == 0 {
		return "", fmt.Errorf("DataFrame has no columns")
	}
	return columns[0], nil
}

func isValidVizType(t string) bool {
	for _, v := range validVizTypes {
		if v == t {
			return true
		}
	}
	return false
}

package viz

import (
	"fmt"
	"strings"

	"github.com/charmbracelet/lipgloss"
	"github.com/vchitepu/gopandas/lib/dataframe"
	"github.com/vchitepu/gopandas/lib/dtype"
)

const minColWidth = 6

// RenderTable renders a styled box-drawing table from a DataFrame.
func RenderTable(df dataframe.DataFrame, th Theme, termWidth int) string {
	rows, _ := df.Shape()
	columns := df.Columns()
	dtypes := df.DTypes()

	if len(columns) == 0 || rows == 0 {
		return "Empty DataFrame"
	}

	// Collect cell strings: header + data rows.
	// For long tables (>50 rows), show first 25 + omission + last 25.
	showFirst := rows
	showLast := 0
	omitted := 0
	if rows > 50 {
		showFirst = 25
		showLast = 25
		omitted = rows - showFirst - showLast
	}

	// Build header cells
	headerCells := make([]string, len(columns))
	copy(headerCells, columns)

	// Build data rows
	var dataRows [][]string
	for i := 0; i < showFirst; i++ {
		dataRows = append(dataRows, rowCells(df, i, columns))
	}
	for i := rows - showLast; i < rows; i++ {
		dataRows = append(dataRows, rowCells(df, i, columns))
	}

	// Compute column widths
	colWidths := make([]int, len(columns))
	for i, h := range headerCells {
		if len(h) > colWidths[i] {
			colWidths[i] = len(h)
		}
	}
	for _, row := range dataRows {
		for i, cell := range row {
			if i < len(colWidths) && len(cell) > colWidths[i] {
				colWidths[i] = len(cell)
			}
		}
	}

	// Fit to terminal width
	colWidths = fitWidths(colWidths, termWidth)

	// Determine alignment per column
	rightAlign := make([]bool, len(columns))
	for i, col := range columns {
		d := dtypes[col]
		rightAlign[i] = d == dtype.Int64 || d == dtype.Float64
	}

	bStyle := lipgloss.NewStyle().Foreground(th.TableBorder)

	var buf strings.Builder

	// Top border
	buf.WriteString(hSep(colWidths, bStyle, "┌", "┬", "┐"))
	buf.WriteByte('\n')

	// Header row
	buf.WriteString(bStyle.Render("│"))
	for i, h := range headerCells {
		cell := truncateCell(h, colWidths[i])
		padded := padCell(cell, colWidths[i], false) // headers left-aligned
		styled := th.TableHeader.Render(padded)
		buf.WriteString(" " + styled + " " + bStyle.Render("│"))
	}
	buf.WriteByte('\n')

	// Header separator
	buf.WriteString(hSep(colWidths, bStyle, "├", "┼", "┤"))
	buf.WriteByte('\n')

	// First N data rows
	for ri, row := range dataRows {
		if omitted > 0 && ri == showFirst {
			// Insert omission indicator
			totalWidth := tableTotalWidth(colWidths)
			msg := fmt.Sprintf(" … (%d rows omitted) … ", omitted)
			if len(msg) > totalWidth-2 {
				msg = " … "
			}
			padded := msg + strings.Repeat(" ", totalWidth-len(msg)-2)
			buf.WriteString(bStyle.Render("│") + lipgloss.NewStyle().Faint(true).Render(padded) + bStyle.Render("│"))
			buf.WriteByte('\n')
		}
		buf.WriteString(bStyle.Render("│"))
		for i := 0; i < len(columns); i++ {
			cell := ""
			if i < len(row) {
				cell = row[i]
			}
			cell = truncateCell(cell, colWidths[i])
			padded := padCell(cell, colWidths[i], rightAlign[i])
			buf.WriteString(" " + padded + " " + bStyle.Render("│"))
		}
		buf.WriteByte('\n')
	}

	// Bottom border
	buf.WriteString(hSep(colWidths, bStyle, "└", "┴", "┘"))
	buf.WriteByte('\n')

	return buf.String()
}

// rowCells extracts string values for a single row from the DataFrame.
func rowCells(df dataframe.DataFrame, rowIdx int, columns []string) []string {
	cells := make([]string, len(columns))
	for i, col := range columns {
		val, err := df.At(rowIdx, col)
		if err != nil {
			cells[i] = ""
			continue
		}
		if val == nil {
			cells[i] = "<null>"
		} else {
			cells[i] = fmt.Sprintf("%v", val)
		}
	}
	return cells
}

// hSep builds a horizontal separator line: ┌──┬──┐ / ├──┼──┤ / └──┴──┘
func hSep(colWidths []int, bStyle lipgloss.Style, left, mid, right string) string {
	parts := make([]string, len(colWidths))
	for i, w := range colWidths {
		parts[i] = strings.Repeat("─", w+2)
	}
	return bStyle.Render(left + strings.Join(parts, mid) + right)
}

// padCell pads a cell to width, right-aligning if ra is true.
func padCell(text string, width int, ra bool) string {
	gap := width - len(text)
	if gap < 0 {
		gap = 0
	}
	if ra {
		return strings.Repeat(" ", gap) + text
	}
	return text + strings.Repeat(" ", gap)
}

// truncateCell truncates text to width, appending "…" if needed.
func truncateCell(text string, width int) string {
	if len(text) <= width {
		return text
	}
	if width <= 1 {
		return "…"
	}
	return text[:width-1] + "…"
}

// fitWidths shrinks column widths so the table fits within termWidth.
// Algorithm: repeatedly shrink the widest column by 1 until it fits.
func fitWidths(colWidths []int, termWidth int) []int {
	total := tableTotalWidth(colWidths)
	for total > termWidth {
		idx := -1
		maxW := -1
		for i, w := range colWidths {
			if w > maxW && w > minColWidth {
				maxW = w
				idx = i
			}
		}
		if idx == -1 {
			break // all columns at minimum
		}
		colWidths[idx]--
		total--
	}
	return colWidths
}

// tableTotalWidth computes the full rendered width of the table.
// Each column gets: 1 space + content + 1 space, plus n+1 border chars.
func tableTotalWidth(colWidths []int) int {
	sum := 0
	for _, w := range colWidths {
		sum += w
	}
	n := len(colWidths)
	return sum + (3 * n) + 1
}

// RenderBar renders a horizontal bar chart.
// XCol provides labels, YCol provides values.
func RenderBar(df dataframe.DataFrame, opts ChartOptions, th Theme, termWidth int) string {
	rows, _ := df.Shape()
	if rows == 0 {
		return "No data to chart"
	}

	// Extract labels and values
	labels := make([]string, rows)
	values := make([]float64, rows)
	maxLabel := 0
	for i := 0; i < rows; i++ {
		lv, _ := df.At(i, opts.XCol)
		if lv == nil {
			labels[i] = "<null>"
		} else {
			labels[i] = fmt.Sprintf("%v", lv)
		}
		if len(labels[i]) > maxLabel {
			maxLabel = len(labels[i])
		}

		yv, _ := df.At(i, opts.YCol)
		values[i] = toFloat64(yv)
	}

	// Find value range
	maxVal := math.Inf(-1)
	minVal := math.Inf(1)
	for _, v := range values {
		if v > maxVal {
			maxVal = v
		}
		if v < minVal {
			minVal = v
		}
	}
	absMax := math.Max(math.Abs(maxVal), math.Abs(minVal))
	if absMax == 0 {
		absMax = 1
	}

	// Max bar width: terminal - label - gutter (4 chars: 2 spaces + value space)
	valueWidth := 12 // space for formatted value
	barMaxWidth := termWidth - maxLabel - 2 - valueWidth
	if barMaxWidth < 5 {
		barMaxWidth = 5
	}

	barStyle := lipgloss.NewStyle().Foreground(th.BarColor)
	negStyle := lipgloss.NewStyle().Foreground(th.BarColor).Faint(true)
	valStyle := lipgloss.NewStyle().Faint(true)

	var buf strings.Builder

	// Title
	if opts.Title != "" {
		buf.WriteString(th.Bold.Render(opts.Title))
		buf.WriteByte('\n')
		buf.WriteByte('\n')
	}

	for i := 0; i < rows; i++ {
		// Pad label
		label := labels[i]
		label = label + strings.Repeat(" ", maxLabel-len(label))

		// Compute bar length
		v := values[i]
		valStr := formatFloat(v)

		var bar string
		if v == 0 {
			bar = barStyle.Render("▏")
		} else if v > 0 {
			barLen := int(math.Round(math.Abs(v) / absMax * float64(barMaxWidth)))
			if barLen < 1 {
				barLen = 1
			}
			bar = barStyle.Render(strings.Repeat("█", barLen))
		} else {
			barLen := int(math.Round(math.Abs(v) / absMax * float64(barMaxWidth)))
			if barLen < 1 {
				barLen = 1
			}
			bar = negStyle.Render(strings.Repeat("░", barLen))
		}

		buf.WriteString(label + "  " + bar + " " + valStyle.Render(valStr))
		buf.WriteByte('\n')
	}

	return buf.String()
}

func toFloat64(v any) float64 {
	switch n := v.(type) {
	case float64:
		return n
	case float32:
		return float64(n)
	case int:
		return float64(n)
	case int64:
		return float64(n)
	case int32:
		return float64(n)
	case int16:
		return float64(n)
	case int8:
		return float64(n)
	default:
		return 0
	}
}

	if v == math.Trunc(v) && math.Abs(v) < 1e15 {
		return fmt.Sprintf("%.0f", v)
	}
	return fmt.Sprintf("%.2f", v)
}

// blockChars are 8 Unicode block levels from lowest to highest.
var blockChars = []rune{'▁', '▂', '▃', '▄', '▅', '▆', '▇', '█'}

// RenderHistogram renders a vertical histogram with 8-level Unicode blocks.
func RenderHistogram(df dataframe.DataFrame, opts ChartOptions, th Theme, termWidth int) string {
	rows, _ := df.Shape()
	if rows == 0 {
		return "No data to chart"
	}

	bins := opts.Bins
	if bins <= 0 {
		bins = 10
	}

	// Extract numeric values
	values := make([]float64, 0, rows)
	for i := 0; i < rows; i++ {
		v, _ := df.At(i, opts.XCol)
		f := toFloat64(v)
		values = append(values, f)
	}

	// Find min/max
	minVal := values[0]
	maxVal := values[0]
	for _, v := range values[1:] {
		if v < minVal {
			minVal = v
		}
		if v > maxVal {
			maxVal = v
		}
	}

	// Compute bin edges and counts
	binWidth := (maxVal - minVal) / float64(bins)
	if binWidth == 0 {
		binWidth = 1
	}
	counts := make([]int, bins)
	for _, v := range values {
		idx := int((v - minVal) / binWidth)
		if idx >= bins {
			idx = bins - 1
		}
		if idx < 0 {
			idx = 0
		}
		counts[idx]++
	}

	// Find max count for scaling
	maxCount := 0
	for _, c := range counts {
		if c > maxCount {
			maxCount = c
		}
	}
	if maxCount == 0 {
		maxCount = 1
	}

	chartHeight := 10 // number of text rows for the chart

	barStyle := lipgloss.NewStyle().Foreground(th.BarColor)
	axisStyle := th.AxisColor

	// Y-axis label width
	yLabelW := len(fmt.Sprintf("%d", maxCount))
	if yLabelW < 3 {
		yLabelW = 3
	}

	var buf strings.Builder

	// Title
	if opts.Title != "" {
		buf.WriteString(th.Bold.Render(opts.Title))
		buf.WriteByte('\n')
		buf.WriteByte('\n')
	}

	// Render rows from top to bottom
	for row := chartHeight; row >= 1; row-- {
		threshold := float64(row) / float64(chartHeight) * float64(maxCount)

		// Y-axis label (only on a few rows)
		yLabel := ""
		if row == chartHeight {
			yLabel = fmt.Sprintf("%*d", yLabelW, maxCount)
		} else if row == 1 {
			yLabel = fmt.Sprintf("%*d", yLabelW, 0)
		} else if row == chartHeight/2 {
			yLabel = fmt.Sprintf("%*d", yLabelW, maxCount/2)
		} else {
			yLabel = strings.Repeat(" ", yLabelW)
		}
		buf.WriteString(axisStyle.Render(yLabel) + " ")

		// Bin columns
		for _, c := range counts {
			frac := float64(c) / float64(maxCount) * float64(chartHeight)
			if frac >= float64(row) {
				// Full block
				buf.WriteString(barStyle.Render(string(blockChars[7])))
			} else if frac >= float64(row)-1 && frac > float64(row-1) {
				// Partial block — pick level 0..7 based on fractional part
				partial := frac - float64(row-1)
				level := int(partial * 8)
				if level > 7 {
					level = 7
				}
				if level < 0 {
					level = 0
				}
				buf.WriteString(barStyle.Render(string(blockChars[level])))
			} else {
				buf.WriteString(" ")
			}
		}
		buf.WriteByte('\n')
	}

	// X-axis line
	buf.WriteString(strings.Repeat(" ", yLabelW+1))
	buf.WriteString(axisStyle.Render(strings.Repeat("─", bins)))
	buf.WriteByte('\n')

	// X-axis labels (show first, middle, last bin boundaries)
	buf.WriteString(strings.Repeat(" ", yLabelW+1))
	firstLabel := formatFloat(minVal)
	midLabel := formatFloat(minVal + (maxVal-minVal)/2)
	lastLabel := formatFloat(maxVal)

	if gap < 2 {
		// Just show first and last
		gap2 := bins - len(firstLabel) - len(lastLabel)
		if gap2 < 1 {
			gap2 = 1
		}
		buf.WriteString(axisStyle.Render(firstLabel + strings.Repeat(" ", gap2) + lastLabel))
	} else {
		leftGap := (bins - len(midLabel)) / 2 - len(firstLabel)
		if leftGap < 1 {
			leftGap = 1
		}
		rightGap := bins - len(firstLabel) - leftGap - len(midLabel) - len(lastLabel)
		if rightGap < 1 {

// Braille dot offsets. Each Braille char is a 2-col × 4-row grid.
// Left column bits: row0=0x01, row1=0x02, row2=0x04, row3=0x40
// Right column bits: row0=0x08, row1=0x10, row2=0x20, row3=0x80
var brailleDots = [4][2]rune{
	{0x01, 0x08}, // row 0 (top)
	{0x02, 0x10}, // row 1
	{0x04, 0x20}, // row 2
	{0x40, 0x80}, // row 3 (bottom)
}

const brailleBase = 0x2800

// RenderLine renders a Braille line chart.
func RenderLine(df dataframe.DataFrame, opts ChartOptions, th Theme, termWidth int) string {
	rows, _ := df.Shape()
	if rows == 0 {
		return "No data to chart"
	}

	// Extract and sort data points by x
	type point struct{ x, y float64 }
	points := make([]point, 0, rows)
	for i := 0; i < rows; i++ {
		xv, _ := df.At(i, opts.XCol)
		yv, _ := df.At(i, opts.YCol)
		points = append(points, point{x: toFloat64(xv), y: toFloat64(yv)})
	}
	// Sort by x
	sortPoints(points)

	// Find data range
	xMin, xMax := points[0].x, points[0].x
	yMin, yMax := points[0].y, points[0].y
	for _, p := range points[1:] {
		if p.x < xMin { xMin = p.x }
		if p.x > xMax { xMax = p.x }
		if p.y < yMin { yMin = p.y }
		if p.y > yMax { yMax = p.y }
	}
	xRange := xMax - xMin
	yRange := yMax - yMin
	if xRange == 0 { xRange = 1 }
	if yRange == 0 { yRange = 1 }

	// Y-axis label width
	yLabelW := len(formatFloat(yMax))
	yMinLabelW := len(formatFloat(yMin))
	if yMinLabelW > yLabelW {
		yLabelW = yMinLabelW
	}
	if yLabelW < 4 {
		yLabelW = 4
	}

	// Grid size in characters
	gridCols := termWidth - yLabelW - 3
	if gridCols < 10 {
		gridCols = 10
	}
	gridRows := 15 // character rows
	// Pixel resolution: each char = 2 cols × 4 rows
	pixelCols := gridCols * 2
	pixelRows := gridRows * 4

	// Build pixel grid (Braille dot accumulator per character cell)
	grid := make([][]rune, gridRows)
	for i := range grid {
		grid[i] = make([]rune, gridCols)
	}

	// Plot points
	for _, p := range points {
		px := int((p.x - xMin) / xRange * float64(pixelCols-1))
		py := int((p.y - yMin) / yRange * float64(pixelRows-1))
		// Invert y (top = high values)
		py = pixelRows - 1 - py

		// Map pixel to character cell
		charCol := px / 2
		charRow := py / 4
		dotCol := px % 2
		dotRow := py % 4

		if charCol >= 0 && charCol < gridCols && charRow >= 0 && charRow < gridRows {
			grid[charRow][charCol] |= brailleDots[dotRow][dotCol]
		}
	}

	// Also draw line segments between consecutive points
	for i := 0; i < len(points)-1; i++ {
		p0 := points[i]
		p1 := points[i+1]
		px0 := int((p0.x - xMin) / xRange * float64(pixelCols-1))
		py0 := pixelRows - 1 - int((p0.y - yMin) / yRange * float64(pixelRows-1))
		px1 := int((p1.x - xMin) / xRange * float64(pixelCols-1))
		py1 := pixelRows - 1 - int((p1.y - yMin) / yRange * float64(pixelRows-1))
		plotLine(grid, gridCols, gridRows, px0, py0, px1, py1)
	}

	lineStyle := lipgloss.NewStyle().Foreground(th.BarColor)
	axisStyle := th.AxisColor

	var buf strings.Builder

	// Title
	if opts.Title != "" {
		buf.WriteString(th.Bold.Render(opts.Title))
		buf.WriteByte('\n')
		buf.WriteByte('\n')
	}

	// Render grid with y-axis labels
	for row := 0; row < gridRows; row++ {
		// Y-axis label
		yLabel := ""
		if row == 0 {
			yLabel = fmt.Sprintf("%*s", yLabelW, formatFloat(yMax))
		} else if row == gridRows-1 {
			yLabel = fmt.Sprintf("%*s", yLabelW, formatFloat(yMin))
		} else if row == gridRows/2 {
			mid := yMin + yRange/2
			yLabel = fmt.Sprintf("%*s", yLabelW, formatFloat(mid))
		} else {
			yLabel = strings.Repeat(" ", yLabelW)
		}
		buf.WriteString(axisStyle.Render(yLabel) + " " + axisStyle.Render("│"))

		// Braille characters
		for col := 0; col < gridCols; col++ {
			ch := brailleBase + grid[row][col]
			buf.WriteString(lineStyle.Render(string(ch)))
		}
		buf.WriteByte('\n')
	}

	// X-axis line
	buf.WriteString(strings.Repeat(" ", yLabelW) + " " + axisStyle.Render("└"+strings.Repeat("─", gridCols)))
	buf.WriteByte('\n')

	// X-axis labels
	xMinLabel := formatFloat(xMin)
	xMaxLabel := formatFloat(xMax)
	xMidLabel := formatFloat(xMin + xRange/2)
	buf.WriteString(strings.Repeat(" ", yLabelW+2))
	gap := gridCols - len(xMinLabel) - len(xMidLabel) - len(xMaxLabel)
	if gap >= 2 {
		leftGap := (gridCols - len(xMidLabel)) / 2 - len(xMinLabel)
		if leftGap < 1 { leftGap = 1 }
		rightGap := gridCols - len(xMinLabel) - leftGap - len(xMidLabel) - len(xMaxLabel)
		if rightGap < 1 { rightGap = 1 }
		buf.WriteString(axisStyle.Render(xMinLabel + strings.Repeat(" ", leftGap) + xMidLabel + strings.Repeat(" ", rightGap) + xMaxLabel))
	} else {
		gap2 := gridCols - len(xMinLabel) - len(xMaxLabel)
		if gap2 < 1 { gap2 = 1 }
		buf.WriteString(axisStyle.Render(xMinLabel + strings.Repeat(" ", gap2) + xMaxLabel))
	}
	buf.WriteByte('\n')

	return buf.String()
}

// sortPoints sorts a slice of points by x using insertion sort (small data).
func sortPoints(pts []point) {
	for i := 1; i < len(pts); i++ {
		key := pts[i]
		j := i - 1
		for j >= 0 && pts[j].x > key.x {
			pts[j+1] = pts[j]
			j--
		}
		pts[j+1] = key
	}
}

// plotLine draws a line between two pixel coordinates using Bresenham's algorithm.
func plotLine(grid [][]rune, gridCols, gridRows, x0, y0, x1, y1 int) {
	dx := x1 - x0
	dy := y1 - y0
	if dx < 0 { dx = -dx }
	if dy < 0 { dy = -dy }
	sx := 1
	if x0 > x1 { sx = -1 }
	sy := 1
	if y0 > y1 { sy = -1 }
	err := dx - dy

	for {
		// Set pixel
		charCol := x0 / 2
		charRow := y0 / 4
		dotCol := x0 % 2
		dotRow := y0 % 4
		if charCol >= 0 && charCol < gridCols && charRow >= 0 && charRow < gridRows &&
			dotCol >= 0 && dotCol < 2 && dotRow >= 0 && dotRow < 4 {
			grid[charRow][charCol] |= brailleDots[dotRow][dotCol]
		}

		if x0 == x1 && y0 == y1 {
			break
		}
		e2 := 2 * err
		if e2 > -dy {
			err -= dy
			x0 += sx
		}
		if e2 < dx {
			err += dx
			y0 += sy
		}
	}
}

}

// sectionSep builds a section separator: ├─ LABEL ──…─┤
func sectionSep(bStyle lipgloss.Style, sectionStyle lipgloss.Style, label string, panelWidth int) string {
	styledLabel := " " + sectionStyle.Render(label) + " "
	// The styled label has ANSI codes, so use raw label len for fill calculation.
	rawLen := len(label) + 2 // spaces around label
	fill := panelWidth - 1 - rawLen
	if fill < 1 {
		fill = 1
	}
	return bStyle.Render("├─") + styledLabel + bStyle.Render(strings.Repeat("─", fill)+"┤")
}

// padRight pads a string to at least width characters. If the visible length
// of text is shorter than width, spaces are appended.
func padRight(text string, width int) string {
	// Count visible characters (strip ANSI codes for measurement).
	visLen := visibleLen(text)
	if visLen >= width {
		return text
	}
	return text + strings.Repeat(" ", width-visLen)
}

// visibleLen returns the visible character count of a string,
// stripping ANSI escape sequences.
func visibleLen(s string) int {
	count := 0
	inEsc := false
	for _, r := range s {
		if inEsc {
			if (r >= 'a' && r <= 'z') || (r >= 'A' && r <= 'Z') {
				inEsc = false
			}
			continue
		}
		if r == '\x1b' {
			inEsc = true
			continue
		}
		count++
	}
	return count
}
```

**Note on `df.Describe()` output:** The `Describe()` method returns a new DataFrame with rows named "count", "mean", "std", "min", "max" and only numeric columns. The `RenderTable` function is reused to render both the statistics sub-table and the preview sub-table, keeping the implementation DRY.

The `padRight` function uses `visibleLen` to correctly pad strings that contain ANSI escape codes from lipgloss styling, ensuring the outer panel borders align properly.
