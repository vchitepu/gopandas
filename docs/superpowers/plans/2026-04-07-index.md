Create `index/index.go`:

```go
package index

// Index is the row label abstraction for gopandas.
// Every Series and DataFrame has an Index that maps labels to positional offsets.
type Index interface {
	// Len returns the number of labels in this index.
	Len() int

	// Labels returns all labels as a slice of any.
	Labels() []any

	// Loc returns the positional offset for the given label.
	// Returns (offset, true) if found, (-1, false) if not.
	Loc(label any) (int, bool)

	// Slice returns a new Index containing labels from position start to end (exclusive).
	Slice(start, end int) Index

	// Name returns the name of this index (may be empty).
	Name() string
}
```

Create `index/range.go`:

```go
package index

import "fmt"

// RangeIndex is the default index type, representing integer labels 0..n-1.
type RangeIndex struct {
	len  int
	name string
}

// NewRangeIndex creates a RangeIndex with n labels (0 through n-1).
// Panics if n < 0.
func NewRangeIndex(n int, name string) *RangeIndex {
	if n < 0 {
		panic(fmt.Sprintf("index: NewRangeIndex called with negative length %d", n))
	}
	return &RangeIndex{len: n, name: name}
}

func (r *RangeIndex) Len() int { return r.len }

func (r *RangeIndex) Labels() []any {
	out := make([]any, r.len)
	for i := range out {
		out[i] = i
	}
	return out
}

func (r *RangeIndex) Loc(label any) (int, bool) {
	v, ok := label.(int)
	if !ok {
		return -1, false
	}
	if v < 0 || v >= r.len {
		return -1, false
	}
	return v, true
}

func (r *RangeIndex) Slice(start, end int) Index {
	if start < 0 {
		start = 0
	}
	if end > r.len {
		end = r.len
	}
	if start >= end {
		return NewRangeIndex(0, r.name)
	}
	return NewRangeIndex(end-start, r.name)
}

func (r *RangeIndex) Name() string { return r.name }
```

```go
package index

// StringIndex is an index backed by string labels.
type StringIndex struct {
	labels []string
	lookup map[string]int // label -> first occurrence position
	name   string
}

// NewStringIndex creates a StringIndex from the given labels.
// If duplicate labels exist, Loc returns the position of the first occurrence.
func NewStringIndex(labels []string, name string) *StringIndex {
	cp := make([]string, len(labels))
	copy(cp, labels)

	lookup := make(map[string]int, len(labels))
	for i, l := range cp {
		if _, exists := lookup[l]; !exists {
			lookup[l] = i
		}
	}
	return &StringIndex{labels: cp, lookup: lookup, name: name}
}

func (s *StringIndex) Len() int { return len(s.labels) }

func (s *StringIndex) Labels() []any {
	out := make([]any, len(s.labels))
	for i, l := range s.labels {
		out[i] = l
	}
	return out
}

func (s *StringIndex) Loc(label any) (int, bool) {
	v, ok := label.(string)
	if !ok {
		return -1, false
	}
	pos, found := s.lookup[v]
	if !found {
		return -1, false
	}
	return pos, true
}

func (s *StringIndex) Slice(start, end int) Index {
	if start < 0 {
		start = 0
	}
	if end > len(s.labels) {
		end = len(s.labels)
	}
	if start >= end {
		return NewStringIndex(nil, s.name)
	}
	return NewStringIndex(s.labels[start:end], s.name)
}

func (s *StringIndex) Name() string { return s.name }
```

```go
package index

import "fmt"

// MultiIndex is a hierarchical index composed of multiple levels.
// Each level is a []any of the same length. A row's key is the tuple
// of values across all levels at that position.
type MultiIndex struct {
	levels [][]any // levels[levelIdx][rowIdx]
	nrows  int
	name   string
}

// NewMultiIndex creates a MultiIndex from the given levels.
// All level slices must have the same length.
// Panics if levels is empty or if level lengths are inconsistent.
func NewMultiIndex(levels [][]any, name string) *MultiIndex {
	if len(levels) == 0 {
		panic("index: NewMultiIndex called with no levels")
	}
	nrows := len(levels[0])
	for i, lv := range levels {
		if len(lv) != nrows {
			panic(fmt.Sprintf(
				"index: NewMultiIndex level %d has length %d, expected %d",
				i, len(lv), nrows,
			))
		}
	}

	// Deep copy levels
	cp := make([][]any, len(levels))
	for i, lv := range levels {
		cp[i] = make([]any, len(lv))
		copy(cp[i], lv)
	}

	return &MultiIndex{levels: cp, nrows: nrows, name: name}
}

func (m *MultiIndex) Len() int { return m.nrows }

// Labels returns one []any per row, where each element is itself a []any
// tuple representing that row's key across all levels.
func (m *MultiIndex) Labels() []any {
	out := make([]any, m.nrows)
	for row := 0; row < m.nrows; row++ {
		tuple := make([]any, len(m.levels))
		for lv := range m.levels {
			tuple[lv] = m.levels[lv][row]
		}
		out[row] = tuple
	}
	return out
}

// Loc finds the first row whose key tuple matches the given label.
// The label must be a []any with length equal to the number of levels.
// Returns (-1, false) if no match or wrong type/length.
func (m *MultiIndex) Loc(label any) (int, bool) {
	key, ok := label.([]any)
	if !ok {
		return -1, false
	}
	if len(key) != len(m.levels) {
		return -1, false
	}
	for row := 0; row < m.nrows; row++ {
		match := true
		for lv := range m.levels {
			if m.levels[lv][row] != key[lv] {
				match = false
				break
			}
		}
		if match {
			return row, true
		}
	}
	return -1, false
}

func (m *MultiIndex) Slice(start, end int) Index {
	if start < 0 {
		start = 0
	}
	if end > m.nrows {
		end = m.nrows
	}
	if start >= end {
		// Return empty MultiIndex with same number of levels
		empty := make([][]any, len(m.levels))
		for i := range empty {
			empty[i] = []any{}
		}
		return NewMultiIndex(empty, m.name)
	}
	sliced := make([][]any, len(m.levels))
	for i, lv := range m.levels {
		sliced[i] = lv[start:end]
	}
	return NewMultiIndex(sliced, m.name)
}

func (m *MultiIndex) Name() string { return m.name }
```
