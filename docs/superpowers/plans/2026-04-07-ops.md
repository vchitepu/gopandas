```go
// JoinType specifies the type of join for Merge and Join operations.
type JoinType int

const (
	Inner JoinType = iota
	Left
	Right
	Outer
)

// String returns the string representation of a JoinType.
func (jt JoinType) String() string {
	switch jt {
	case Inner:
		return "inner"
	case Left:
		return "left"
	case Right:
		return "right"
	case Outer:
		return "outer"
	default:
		return "unknown"
	}
}
```

### Step 2.3: Implement Merge with inner join support

- [ ] Create `ops/merge.go`:

```go
package ops

import (
	"fmt"
	"strings"

	"github.com/vinaychitepu/gopandas/dataframe"
)

// Merge performs a SQL-style join on one or more key columns.
// The `on` columns must exist in both left and right DataFrames.
// Supported join types: Inner, Left, Right, Outer.
func Merge(left, right dataframe.DataFrame, on []string, how JoinType) (dataframe.DataFrame, error) {
	if len(on) == 0 {
		return dataframe.DataFrame{}, fmt.Errorf("ops.Merge: on must specify at least one column")
	}

	// Validate that key columns exist in both DataFrames
	for _, col := range on {
		if _, err := left.Col(col); err != nil {
			return dataframe.DataFrame{}, fmt.Errorf("ops.Merge: left DataFrame: %w", err)
		}
		if _, err := right.Col(col); err != nil {
			return dataframe.DataFrame{}, fmt.Errorf("ops.Merge: right DataFrame: %w", err)
		}
	}

	// Build key map for right DataFrame: composite key -> []row indices
	rightKeyMap := buildKeyMap(right, on)

	// Determine output columns: key columns + left non-key + right non-key
	leftCols := nonKeyCols(left.Columns(), on)
	rightCols := nonKeyCols(right.Columns(), on)

	// Build row pairs: (leftRow, rightRow) for matched rows
	type rowPair struct {
		leftRow  int
		rightRow int // -1 means no match (for left/outer joins)
	}
	var pairs []rowPair

	for i := 0; i < left.Len(); i++ {
		leftKey := extractKey(left, on, i)
		if rightRows, ok := rightKeyMap[leftKey]; ok {
			for _, rRow := range rightRows {
				pairs = append(pairs, rowPair{leftRow: i, rightRow: rRow})
			}
		} else {
			switch how {
			case Left, Outer:
				pairs = append(pairs, rowPair{leftRow: i, rightRow: -1})
			case Inner, Right:
				// skip — no match
			}
		}
	}

	// For Right and Outer joins: add unmatched right rows
	if how == Right || how == Outer {
		leftKeyMap := buildKeyMap(left, on)
		for j := 0; j < right.Len(); j++ {
			rightKey := extractKey(right, on, j)
			if _, ok := leftKeyMap[rightKey]; !ok {
				pairs = append(pairs, rowPair{leftRow: -1, rightRow: j})
			}
		}
	}

	// Assemble the result columns
	nRows := len(pairs)
	result := make(map[string]any)

	// Key columns — take from whichever side has the value
	for _, col := range on {
		vals := make([]any, nRows)
		for i, p := range pairs {
			if p.leftRow >= 0 {
				v, _ := left.At(p.leftRow, col)
				vals[i] = v
			} else {
				v, _ := right.At(p.rightRow, col)
				vals[i] = v
			}
		}
		result[col] = vals
	}

	// Left non-key columns
	for _, col := range leftCols {
		vals := make([]any, nRows)
		for i, p := range pairs {
			if p.leftRow >= 0 {
				v, _ := left.At(p.leftRow, col)
				vals[i] = v
			} else {
				vals[i] = nil
			}
		}
		result[col] = vals
	}

	// Right non-key columns
	for _, col := range rightCols {
		vals := make([]any, nRows)
		for i, p := range pairs {
			if p.rightRow >= 0 {
				v, _ := right.At(p.rightRow, col)
				vals[i] = v
			} else {
				vals[i] = nil
			}
		}
		result[col] = vals
	}

	return dataframe.FromRecords(buildRecords(result, nRows, on, leftCols, rightCols))
}

// buildKeyMap creates a map from composite key string -> list of row indices.
func buildKeyMap(df dataframe.DataFrame, keyCols []string) map[string][]int {
	m := make(map[string][]int)
	for i := 0; i < df.Len(); i++ {
		key := extractKey(df, keyCols, i)
		m[key] = append(m[key], i)
	}
	return m
}

// extractKey builds a composite key string for row i from the given columns.
func extractKey(df dataframe.DataFrame, keyCols []string, row int) string {
	parts := make([]string, len(keyCols))
	for j, col := range keyCols {
		v, _ := df.At(row, col)
		parts[j] = fmt.Sprintf("%v", v)
	}
	return strings.Join(parts, "|")
}

// nonKeyCols returns columns from `all` that are not in `keys`.
func nonKeyCols(all []string, keys []string) []string {
	keySet := make(map[string]bool, len(keys))
	for _, k := range keys {
		keySet[k] = true
	}
	var out []string
	for _, col := range all {
		if !keySet[col] {
			out = append(out, col)
		}
	}
	return out
}

// buildRecords converts columnar map data to []map[string]any for FromRecords.
func buildRecords(data map[string]any, nRows int, keyCols, leftCols, rightCols []string) []map[string]any {
	if nRows == 0 {
		return []map[string]any{}
	}
	allCols := make([]string, 0, len(keyCols)+len(leftCols)+len(rightCols))
	allCols = append(allCols, keyCols...)
	allCols = append(allCols, leftCols...)
	allCols = append(allCols, rightCols...)

	records := make([]map[string]any, nRows)
	for i := 0; i < nRows; i++ {
		rec := make(map[string]any, len(allCols))
		for _, col := range allCols {
			vals := data[col].([]any)
			rec[col] = vals[i]
		}
		records[i] = rec
	}
	return records
}
```

```go
package ops

import (
	"fmt"

	"github.com/vinaychitepu/gopandas/dataframe"
)

// Join performs an index-based join between two DataFrames.
// Both DataFrames must have their index set (e.g., via SetIndex).
// lsuffix and rsuffix are appended to overlapping non-key column names.
func Join(left, right dataframe.DataFrame, how JoinType, lsuffix, rsuffix string) (dataframe.DataFrame, error) {
	leftIdx := left.Index()
	rightIdx := right.Index()

	// Build a map of right index labels -> row positions
	rightLabelMap := make(map[string][]int)
	rightLabels := rightIdx.Labels()
	for i, lbl := range rightLabels {
		key := fmt.Sprintf("%v", lbl)
		rightLabelMap[key] = append(rightLabelMap[key], i)
	}

	leftLabelMap := make(map[string][]int)
	leftLabels := leftIdx.Labels()
	for i, lbl := range leftLabels {
		key := fmt.Sprintf("%v", lbl)
		leftLabelMap[key] = append(leftLabelMap[key], i)
	}

	// Determine overlapping column names
	leftColSet := make(map[string]bool)
	for _, c := range left.Columns() {
		leftColSet[c] = true
	}
	rightColSet := make(map[string]bool)
	for _, c := range right.Columns() {
		rightColSet[c] = true
	}
	overlap := make(map[string]bool)
	for c := range leftColSet {
		if rightColSet[c] {
			overlap[c] = true
		}
	}

	// Build row pairs
	type rowPair struct {
		leftRow  int
		rightRow int
	}
	var pairs []rowPair

	for i, lbl := range leftLabels {
		key := fmt.Sprintf("%v", lbl)
		if rRows, ok := rightLabelMap[key]; ok {
			for _, rRow := range rRows {
				pairs = append(pairs, rowPair{leftRow: i, rightRow: rRow})
			}
		} else {
			switch how {
			case Left, Outer:
				pairs = append(pairs, rowPair{leftRow: i, rightRow: -1})
			}
		}
	}

	if how == Right || how == Outer {
		for j, lbl := range rightLabels {
			key := fmt.Sprintf("%v", lbl)
			if _, ok := leftLabelMap[key]; !ok {
				pairs = append(pairs, rowPair{leftRow: -1, rightRow: j})
			}
		}
	}

	nRows := len(pairs)
	records := make([]map[string]any, nRows)
	for i := range records {
		records[i] = make(map[string]any)
	}

	// Populate left columns
	for _, col := range left.Columns() {
		outName := col
		if overlap[col] {
			outName = col + lsuffix
		}
		for i, p := range pairs {
			if p.leftRow >= 0 {
				v, _ := left.At(p.leftRow, col)
				records[i][outName] = v
			} else {
				records[i][outName] = nil
			}
		}
	}

	// Populate right columns
	for _, col := range right.Columns() {
		outName := col
		if overlap[col] {
			outName = col + rsuffix
		}
		for i, p := range pairs {
			if p.rightRow >= 0 {
				v, _ := right.At(p.rightRow, col)
				records[i][outName] = v
			} else {
				records[i][outName] = nil
			}
		}
	}

	return dataframe.FromRecords(records)
}
```

```go
import (
	"fmt"
	"sort"

	"github.com/vinaychitepu/gopandas/dataframe"
)

// Concat vertically concatenates multiple DataFrames (axis=0).
// Columns are the union of all input DataFrames' columns.
// Missing columns in any DataFrame are filled with nil.
// The result has a fresh RangeIndex (0..n-1).
func Concat(dfs ...dataframe.DataFrame) (dataframe.DataFrame, error) {
	if len(dfs) == 0 {
		return dataframe.DataFrame{}, fmt.Errorf("ops.Concat: must provide at least one DataFrame")
	}

	if len(dfs) == 1 {
		return dfs[0], nil
	}

	// Collect union of all column names, preserving order of first appearance
	colOrder := make([]string, 0)
	colSet := make(map[string]bool)
	for _, df := range dfs {
		for _, col := range df.Columns() {
			if !colSet[col] {
				colSet[col] = true
				colOrder = append(colOrder, col)
			}
		}
	}
	sort.Strings(colOrder)

	// Calculate total rows
	totalRows := 0
	for _, df := range dfs {
		totalRows += df.Len()
	}

	// Build records row by row
	records := make([]map[string]any, 0, totalRows)
	for _, df := range dfs {
		dfCols := make(map[string]bool)
		for _, c := range df.Columns() {
			dfCols[c] = true
		}
		for i := 0; i < df.Len(); i++ {
			rec := make(map[string]any, len(colOrder))
			for _, col := range colOrder {
				if dfCols[col] {
					v, _ := df.At(i, col)
					rec[col] = v
				} else {
					rec[col] = nil
				}
			}
			records = append(records, rec)
		}
	}

	return dataframe.FromRecords(records)
}
```

```go
package ops

import (
	"fmt"
	"math"
	"sort"

	"github.com/vinaychitepu/gopandas/dataframe"
)

// Pivot reshapes a DataFrame from long to wide format.
// - index: column whose unique values become the row labels
// - columns: column whose unique values become the new column headers
// - values: column whose values fill the pivoted cells
//
// Returns an error if (index, columns) pairs are not unique,
// or if any of the named columns do not exist.
func Pivot(df dataframe.DataFrame, index, columns, values string) (dataframe.DataFrame, error) {
	// Validate columns exist
	for _, col := range []string{index, columns, values} {
		if _, err := df.Col(col); err != nil {
			return dataframe.DataFrame{}, fmt.Errorf("ops.Pivot: %w", err)
		}
	}

	// Collect unique index values (row labels) and column values (new column headers)
	indexVals := uniqueValues(df, index)
	colVals := uniqueValues(df, columns)

	// Build a map: (indexVal, colVal) -> value
	type cellKey struct {
		idx string
		col string
	}
	cells := make(map[cellKey]any)

	for i := 0; i < df.Len(); i++ {
		idxVal, _ := df.At(i, index)
		colVal, _ := df.At(i, columns)
		valVal, _ := df.At(i, values)

		ck := cellKey{idx: fmt.Sprintf("%v", idxVal), col: fmt.Sprintf("%v", colVal)}
		if _, exists := cells[ck]; exists {
			return dataframe.DataFrame{}, fmt.Errorf(
				"ops.Pivot: duplicate entry for index=%v, column=%v", idxVal, colVal,
			)
		}
		cells[ck] = valVal
	}

	// Build output records
	records := make([]map[string]any, len(indexVals))
	for i, idxVal := range indexVals {
		rec := map[string]any{
			index: idxVal,
		}
		for _, colVal := range colVals {
			colName := fmt.Sprintf("%v", colVal)
			ck := cellKey{idx: fmt.Sprintf("%v", idxVal), col: colName}
			if v, ok := cells[ck]; ok {
				rec[colName] = v
			} else {
				rec[colName] = nil
			}
		}
		records[i] = rec
	}

	return dataframe.FromRecords(records)
}

// uniqueValues returns the unique values in a column, sorted as strings.
func uniqueValues(df dataframe.DataFrame, col string) []any {
	seen := make(map[string]any)
	var order []string
	for i := 0; i < df.Len(); i++ {
		v, _ := df.At(i, col)
		key := fmt.Sprintf("%v", v)
		if _, ok := seen[key]; !ok {
			seen[key] = v
			order = append(order, key)
		}
	}
	sort.Strings(order)
	result := make([]any, len(order))
	for i, k := range order {
		result[i] = seen[k]
	}
	return result
}
```

```go
// toFloat64 converts a numeric value to float64.
// Returns 0, false if the value is not numeric.
func toFloat64(v any) (float64, bool) {
	switch n := v.(type) {
	case int64:
		return float64(n), true
	case float64:
		return n, true
	case int:
		return float64(n), true
	case int32:
		return float64(n), true
	case float32:
		return float64(n), true
	default:
		return 0, false
	}
}

// aggregate applies an AggFunc to a slice of float64 values.
func aggregate(vals []float64, aggFunc AggFunc) float64 {
	if len(vals) == 0 {
		return math.NaN()
	}
	switch aggFunc {
	case AggSum:
		sum := 0.0
		for _, v := range vals {
			sum += v
		}
		return sum
	case AggMean:
		sum := 0.0
		for _, v := range vals {
			sum += v
		}
		return sum / float64(len(vals))
	case AggCount:
		return float64(len(vals))
	case AggMin:
		m := vals[0]
		for _, v := range vals[1:] {
			if v < m {
				m = v
			}
		}
		return m
	case AggMax:
		m := vals[0]
		for _, v := range vals[1:] {
			if v > m {
				m = v
			}
		}
		return m
	default:
		return math.NaN()
	}
}

// PivotTable reshapes a DataFrame from long to wide format with aggregation.
// Unlike Pivot, duplicate (index, columns) pairs are aggregated using aggFunc.
// Cells with no data are filled with NaN.
func PivotTable(df dataframe.DataFrame, index, columns, values string, aggFunc AggFunc) (dataframe.DataFrame, error) {
	// Validate columns exist
	for _, col := range []string{index, columns, values} {
		if _, err := df.Col(col); err != nil {
			return dataframe.DataFrame{}, fmt.Errorf("ops.PivotTable: %w", err)
		}
	}

	// Collect unique index values and column values
	indexVals := uniqueValues(df, index)
	colVals := uniqueValues(df, columns)

	// Build a map: (indexVal, colVal) -> []float64
	type cellKey struct {
		idx string
		col string
	}
	cells := make(map[cellKey][]float64)

	for i := 0; i < df.Len(); i++ {
		idxVal, _ := df.At(i, index)
		colVal, _ := df.At(i, columns)
		valVal, _ := df.At(i, values)

		f, ok := toFloat64(valVal)
		if !ok {
			return dataframe.DataFrame{}, fmt.Errorf(
				"ops.PivotTable: non-numeric value %v at row %d", valVal, i,
			)
		}

		ck := cellKey{idx: fmt.Sprintf("%v", idxVal), col: fmt.Sprintf("%v", colVal)}
		cells[ck] = append(cells[ck], f)
	}

	// Build output records
	records := make([]map[string]any, len(indexVals))
	for i, idxVal := range indexVals {
		rec := map[string]any{
			index: idxVal,
		}
		for _, colVal := range colVals {
			colName := fmt.Sprintf("%v", colVal)
			ck := cellKey{idx: fmt.Sprintf("%v", idxVal), col: colName}
			rec[colName] = aggregate(cells[ck], aggFunc)
		}
		records[i] = rec
	}

	return dataframe.FromRecords(records)
}
```

```go
package ops

import (
	"fmt"

	"github.com/vinaychitepu/gopandas/dataframe"
)

// Melt unpivots a DataFrame from wide to long format.
//
//   - idVars: columns to keep as identifier variables (repeated for each value var)
//   - valueVars: columns to unpivot into rows
//   - varName: name for the new "variable" column (defaults to "variable" if empty)
//   - valueName: name for the new "value" column (defaults to "value" if empty)
//
// Output has len(idVars)+2 columns and len(df)*len(valueVars) rows.
func Melt(df dataframe.DataFrame, idVars, valueVars []string, varName, valueName string) (dataframe.DataFrame, error) {
	if varName == "" {
		varName = "variable"
	}
	if valueName == "" {
		valueName = "value"
	}

	// Validate all columns exist
	for _, col := range idVars {
		if _, err := df.Col(col); err != nil {
			return dataframe.DataFrame{}, fmt.Errorf("ops.Melt: id column: %w", err)
		}
	}
	for _, col := range valueVars {
		if _, err := df.Col(col); err != nil {
			return dataframe.DataFrame{}, fmt.Errorf("ops.Melt: value column: %w", err)
		}
	}

	nRows := df.Len()
	nValueVars := len(valueVars)
	records := make([]map[string]any, 0, nRows*nValueVars)

	for _, vv := range valueVars {
		for i := 0; i < nRows; i++ {
			rec := make(map[string]any, len(idVars)+2)

			// Copy id columns
			for _, id := range idVars {
				v, _ := df.At(i, id)
				rec[id] = v
			}

			// Set variable name and value
			rec[varName] = vv
			val, _ := df.At(i, vv)
			rec[valueName] = val

			records = append(records, rec)
		}
	}

	return dataframe.FromRecords(records)
}
```

```go
import (
```

```go
// Transpose flips a DataFrame's rows and columns.
// The original column names become values in an "index" column.
// The original row positions (0, 1, 2, ...) become column headers.
func Transpose(df dataframe.DataFrame) (dataframe.DataFrame, error) {
	cols := df.Columns()
	nRows := df.Len()

	records := make([]map[string]any, len(cols))

	for i, colName := range cols {
		rec := map[string]any{
			"index": colName,
		}
		for j := 0; j < nRows; j++ {
			v, _ := df.At(j, colName)
			rec[fmt.Sprintf("%d", j)] = v
		}
		records[i] = rec
	}

	return dataframe.FromRecords(records)
}
```

```go
// Stack converts a wide DataFrame to a long format with (row, column, value) triples.
// Every cell in the DataFrame produces one row in the output.
func Stack(df dataframe.DataFrame) (dataframe.DataFrame, error) {
	cols := df.Columns()
	nRows := df.Len()

	records := make([]map[string]any, 0, nRows*len(cols))

	for i := 0; i < nRows; i++ {
		for _, colName := range cols {
			v, _ := df.At(i, colName)
			records = append(records, map[string]any{
				"row":    int64(i),
				"column": colName,
				"value":  v,
			})
		}
	}

	return dataframe.FromRecords(records)
}
```

```go
// Unstack converts a long-format DataFrame back to wide format.
// The `column` parameter names the column whose unique values become new column headers.
// The remaining non-column, non-value columns are used as row grouping keys.
//
// Expects exactly 3 columns in the DataFrame:
//   - A row identifier column (any column that is not `column` and not the last column)
//   - The `column` column (whose values become headers)
//   - A value column (the last remaining column, whose values fill the cells)
//
// For DataFrames with exactly 3 columns, Unstack automatically identifies
// the row-key, header, and value columns based on the `column` parameter.
func Unstack(df dataframe.DataFrame, column string) (dataframe.DataFrame, error) {
	// Validate the column exists
	if _, err := df.Col(column); err != nil {
		return dataframe.DataFrame{}, fmt.Errorf("ops.Unstack: %w", err)
	}

	allCols := df.Columns()
	if len(allCols) < 3 {
		return dataframe.DataFrame{}, fmt.Errorf("ops.Unstack: need at least 3 columns, got %d", len(allCols))
	}

	// Identify role of each column:
	// - column: the one whose values become headers
	// - value column: the remaining column that is not used as a row key
	// - row key columns: everything else
	//
	// Convention: the last column that is not `column` is the value column.
	var rowKeyCols []string
	var valueCol string
	for i := len(allCols) - 1; i >= 0; i-- {
		if allCols[i] != column {
			valueCol = allCols[i]
			// All other non-column, non-value columns are row keys
			for j := 0; j < len(allCols); j++ {
				if allCols[j] != column && allCols[j] != valueCol {
					rowKeyCols = append(rowKeyCols, allCols[j])
				}
			}
			break
		}
	}

	// Get unique values for the column that becomes headers
	headerVals := uniqueValues(df, column)

	// Group rows by their row key
	type rowKey string
	buildRowKey := func(i int) rowKey {
		key := ""
		for _, rk := range rowKeyCols {
			v, _ := df.At(i, rk)
			key += fmt.Sprintf("%v|", v)
		}
		return rowKey(key)
	}

	// Collect unique row keys in order
	seenRows := make(map[rowKey]int) // rowKey -> index in records
	var records []map[string]any

	for i := 0; i < df.Len(); i++ {
		rk := buildRowKey(i)
		idx, exists := seenRows[rk]
		if !exists {
			idx = len(records)
			seenRows[rk] = idx
			rec := make(map[string]any)
			// Copy row key columns
			for _, rkCol := range rowKeyCols {
				v, _ := df.At(i, rkCol)
				rec[rkCol] = v
			}
			records = append(records, rec)
		}

		// Set the value under the header
		colVal, _ := df.At(i, column)
		val, _ := df.At(i, valueCol)
		headerName := fmt.Sprintf("%v", colVal)
		records[idx][headerName] = val
	}

	// Fill missing header cells with nil
	for _, rec := range records {
		for _, hv := range headerVals {
			headerName := fmt.Sprintf("%v", hv)
			if _, ok := rec[headerName]; !ok {
				rec[headerName] = nil
			}
		}
	}

	return dataframe.FromRecords(records)
}
```
